7def calculateNormal(vec):
8 '''returns the norm for a given 3d vector'''
9 if len(vec) == 3:
10 v1 = vec[1] - vec[0]
11 v2 = vec[2] - vec[0]
12 return np.cross(v1,v2)/np.sqrt(np.cross(v1,v2)[0]**2+np.cross(v1,v2)[1]**2+np.cross(v1,v2)[2]**2)
13 else:
14 print("Mesh-Generation is going wrong. Faces do not have 3 components")
15 return False

179 def calcRotationVec(self,boundaryLoopNodes):
180 '''returns the rotationVector and the angle based on the boundaryLoopNodes.'''
181 openingMean = [np.mean(self.vertices[boundaryLoopNodes[0]][:,0]),np.mean(self.vertices[boundaryLoopNodes[0]][:,1]),np.mean(self.vertices[boundaryLoopNodes[0]][:,2])]
182 overallMean = np.mean(self.vertices)
183 oldOrientationVec=(openingMean-overallMean)/np.linalg.norm(openingMean-overallMean)
184 zVec = [0,0,1]
185 sina = np.linalg.norm(np.cross(oldOrientationVec,zVec))/(np.linalg.norm(oldOrientationVec)*np.linalg.norm(zVec))
186 cosa = np.linalg.norm(np.dot(oldOrientationVec,zVec))/(np.linalg.norm(oldOrientationVec)*np.linalg.norm(zVec))
187 angle = np.arctan2(sina,cosa)
188 rotationVec = np.cross(oldOrientationVec,zVec)/np.linalg.norm(np.cross(oldOrientationVec,zVec))
189 return rotationVec,angle

191 def calc3DRotMatByVec(self,rotationVec,angle):
192 '''returns the rotation matrix calculated from the rotationVector'''
193 uX,uY,uZ = rotationVec
194 tmp1 = np.sin(angle)
195 tmp2 = np.cos(angle)
196 tmp3 = (1-np.cos(angle))
197 rotMat = np.zeros((3,3))
198 rotMat[0][0] = tmp2 + uX*uX*tmp3
199 rotMat[0][1] = uX*uY*tmp3-uZ*tmp1
200 rotMat[0][2] = uX*uZ*tmp3+uY*tmp1
201 rotMat[1][0] = uY*uX*tmp3+uZ*tmp1
202 rotMat[1][1] = tmp2+uY*uY*tmp3
203 rotMat[1][2] = uY*uZ*tmp3-uX*tmp1
204 rotMat[2][0] = uZ*uX*tmp3-uY*tmp1
205 rotMat[2][1] = uZ*uX*tmp3+uX*tmp1
206 rotMat[2][2] = tmp2+uZ*uZ*tmp3
207 return rotMat

231 def get2Dcoordinates(self):
232 '''returns the from 3D to 2D converted vertices'''
233 corods = np.array(self.rotatedCaylinder)
234 minZCylinder = min(corods[:,2])
235 corods[:,2] = corods[:,2]+minZCylinder
236 phiCoord = np.arctan2(corods[:,1],corods[:,0])
237 r = np.sqrt(corods[:,0]**2+corods[:,1]**2)
238 u = (corods[:,2]-np.mean(r))*np.sin(phiCoord)
239 v = (corods[:,2]- np.mean(r))*np.cos(phiCoord)
